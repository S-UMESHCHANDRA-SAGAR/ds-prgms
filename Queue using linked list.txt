{

{

int del();

	rear=NULL;
}

void QULL:: insert()
{
node *temp;
temp=new node;
cout<<"enter the inserted element in to the queue";
cin>>temp->data;
if(front==NULL)
{
temp->next=front;
front=rear=temp;
}
else
{
temp->next=rear->next;
rear->next=temp;
rear=temp;
}
}
int QULL::del()
{
int k;
node *p;
if(front!=NULL)
{
k=front->data;
p=front;
front=front->next;
free(p);
return(k);
}
}

{
node *p;
